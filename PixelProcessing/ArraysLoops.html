Arrays & Loops

Arrays

Evenly spaced addresses in memory. Can store primitives or object references.

Array: Integer

int[] integers = new int[5];
integers[0] = 5;
integers[1] = 3;
integers[2] = 10;
integers[3] = 22;
integers[4] = 1;
integers[5] = 5;

Array: Float

float[] decimals = new float[3];
decimals[0] = 0.2;
decimals[1] = 3.14;
decimals[2] = -55.1241;

Array: String

String[] words1 = {"cat", "dog", "mouse"};
//these are the same
String[] words2 = new String[3];
words2[0] = "cat";
words2[1] = "dog";
words2[2] = "mouse";

Array: Char

String hello1 = "hello";
char[] hello2 = {'h','e','l','l','o'};
char[] hello3 = hello2.toCharArray();
char[] hello4 = new char[2];
hello4[0] = 'h';
hello4[1] = 'i';

Array: Boolean

Boolean[] torf = {true,false,true};
for(int i = 0; i < torf.length; i++) {
  if(torf[i]) {
    print(i);
  }
}

Loops

for #1

int[] nums = {1,3,-6,1,5,-2,-5,4,-1,0};

for(int i = 0; i < nums.length; i++) {
  print(nums[i]);
}

for #2 (foreach)

int[] nums = {1,3,-6,1,5,-2,-5,4,-1,0};

for(int i : nums) {
  print(i);
}

while

int[] nums = {1,3,-6,1,5,-2,-5,4,-1,0};
int counter = 0;
while(counter < nums.length) {
  print(nums[counter]);
  counter++;
}
